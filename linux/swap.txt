# Desabilitar o swap
swapoff -a

# Criar novo arquivo de SWAP
dd if=/dev/zero of=/swapfile bs=1024 count=6M

# Aplicando permiss√µs
chmod 600 /swapfile

# Definir o arquivo criado como Swap
mkswap /swapfile

# Ativar o swapfile
swapon /swapfile
